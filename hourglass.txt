 static int hourglassSum(int[][] arr) {
        int result = 0, max = -999;
       
        for(int i=0;i<arr.length - 2;i++)
        {
            for(int j=0;j<arr.length - 2;j++)
            {
                result = arr[i][j] + arr[i][j+1] + arr[i][j+2] + arr[i+1][j+1] + arr[i+2][j] +                              arr[i+2][j+1] + arr[i+2][j+2];
                if(max<result){
                    max = result;
                }
